## 🔄 자동화 설정

### Notion API 연동 (선택사항)

```javascript
// 자동 진도율 업데이트 예시
const { Client } = require('@notionhq/client');

const notion = new Client({
  auth: process.env.NOTION_TOKEN,
});

async function updateProgress(studentId, weekNumber, progressRate) {
  try {
    await notion.pages.update({
      page_id: studentId,
      properties: {
        '진도율': {
          number: progressRate
        },
        '현재 주차': {
          select: {
            name: `${weekNumber}주차`
          }
        },
        '마지막 업데이트': {
          date: {
            start: new Date().toISOString()
          }
        }
      }
    });
  } catch (error) {
    console.error('진도율 업데이트 실패:', error);
  }
}
```
